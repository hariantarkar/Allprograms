class product
{
	private int id;
	private String name;
	private int qty;
	private int price;
public void setId(int id)]
{
this.id=id;
}
public int getId()
{
return id;
}
public void setName(String name)
{
this.name=name;
}
public String getName()
{
return name;
}
public void setqty(int qty)
{
this.qty=qty;
}
public int getqty()
{
return qty;
}
public void setprice(int price)
{
this.price=price;
}
public int getprice()
{
return price;
}
}
