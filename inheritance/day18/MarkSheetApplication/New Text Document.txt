class student
{
	private int id;
	private String name;
	private String std;
  student()
  {
   }
public student(int id,String name,String std)
{
	this.id=id;
	this.name=name;
	this.std=std;
}
public void setId(int id)
{
	this.id=id;
}
public int getId()
{
	return id;
}
public void setName(String name)
{
	this.name=name;
}
public  String getName()
{
	return name;
}
public void setStd(String std)
{
	this.std=std;
}
public String getStd()
{
	return std;
}
}
